        function switchCategory(key) {

            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));

            document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));

        // Gestione Carrello

        function addToCart(id, name, price) {

            const existingItem = cart.find(item => item.id === id);

            if (existingItem) {

                existingItem.qty += 1;

            } else {

                cart.push({ id, name, price, qty: 1 });

            }

            updateCartState();

            showToast(`Aggiunto: ${name}`);

        }

        function changeQuantity(id, delta) {

            const item = cart.find(i => i.id === id);

            if (!item) return;

            item.qty += delta;

            if (item.qty <= 0) {

                cart = cart.filter(i => i.id !== id);

            }

            updateCartState();

            renderCartItems();

        }

        function updateCartState() {

            const totalItems = cart.reduce((acc, item) => acc + item.qty, 0);

            const totalPrice = cart.reduce((acc, item) => acc + (item.price * item.qty), 0);

            document.getElementById('cartQtyLabel').innerText = `${totalItems} ${totalItems === 1 ? 'articolo' : 'articoli'}`;

            document.getElementById('cartTotalLabel').innerText = `€${totalPrice.toFixed(2)}`;

           

            const openCartBtn = document.getElementById('openCartBtn');

            if (totalItems > 0) {

                openCartBtn.removeAttribute('disabled');

            } else {

                openCartBtn.setAttribute('disabled', 'true');

                toggleCartModal(false);

            }

        }

        function renderCartItems() {

            const container = document.getElementById('cartItemsList');

            const footer = document.getElementById('cartModalFooter');

           

            if (cart.length === 0) {

                footer.style.display = 'none';

                return;

            }

            container.innerHTML = cart.map(item => {

                const priceText = item.price === 0 ? "Prezzo da concordare" : `€${(item.price * item.qty).toFixed(2)}`;

                return `

                    <div class="cart-item">

                        <div class="item-meta">

                            <h4>${item.name}</h4>

                            <p style="color:var(--hint-color); font-size:13px; margin-top:2px;">${item.price > 0 ? '€' + item.price + ' /cad' : ''}</p>

                        </div>

                        <div class="item-controls">

                            <button class="control-btn" onclick="changeQuantity('${item.id}', -1)">−</button>

                            <span style="font-weight:700; min-width:20px; text-align:center;">${item.qty}</span>

                            <button class="control-btn" onclick="changeQuantity('${item.id}', 1)">+</button>

                            <button class="control-btn delete" onclick="changeQuantity('${item.id}', -${item.qty})">×</button>

                        </div>

                    </div>

                `;

            }).join('');

            document.getElementById('modalTotalLabel').innerText = `€${totalPrice.toFixed(2)}`;

        // Modal Utilities

        function toggleCartModal(show) {

            const modal = document.getElementById('cartModal');

            if (show) {

                renderCartItems();

                modal.classList.add('active');

            } else {

                modal.classList.remove('active');

            }

        }

        function closeModalOnOverlay(e) {

            if (e.target.id === 'cartModal') toggleCartModal(false);

        }
